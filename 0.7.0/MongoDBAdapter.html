Example
import {DS} from 'js-data'
import MongoDBAdapter from 'js-data-mongodb'
const store = new DS()
const adapter = new MongoDBAdapter({
  uri: 'mongodb://localhost:27017'
})
store.registerAdapter('mongodb', adapter, { 'default': true })
